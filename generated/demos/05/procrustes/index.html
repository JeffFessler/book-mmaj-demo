Fitting function

function procrustes(A, B)
    C = B * A'
    (U,s,V) = svd(C)
    Q = U*V'
    scale = sum(s) / norm(A,2)^2
    return Q, scale
end

Explore additional special cases

three points along a line, symmetrical

A = [-1 0 1; 0 0 0]
B = [0 0 0; -2 0 2]
Q, scale = procrustes(A, B)
([-0.0 -1.0; 1.0 0.0], 1.9999999999999996)

check

@assert B ≈ scale * Q * A
